\subsection{Fonctionnalit\'es impl\'ement\'ees}
\frame
{
\frametitle{Fonctionnalit\'es impl\'ement\'ees}
\framesubtitle{Fonctionnalit\'es du programme final}
\begin{itemize}
 \item<1-7> R\'ecup\'eration d'informations sur un AS
 \item<2-7> Calcul du nombre de cliques maximum dans le graphe gr\^ace \`a l'algorithme de Bron and Kerbosch
 \item<3-7> Chargement d'un fichier de triplets pour \'eliminer les stubs du graphe
 \item<4-7> Zoomer sur le proche voisinage d'un AS
 \item<5-7> Revenir au graphe de base
 \item<6-7> Calculer la centralit\'e de Freeman des sommets
 \item<7> Filtrer les sommets selon leur centralit\'e
 \item<7> Filtrer les sommets selon leur num\'ero d'AS
\end{itemize}
